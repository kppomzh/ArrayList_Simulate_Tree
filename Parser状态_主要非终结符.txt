文法
开始符号设置为S，推导用:代替箭头
全大写是非终结符，含有小写的都是终结符，特殊情况是标记常量或变量的QUANTIFIER，里面的终结符也是大写

S:CREATESQL
S:DROPSQL
S:SELECTSQL
S:INSERTSQL
S:DELETESQL
S:UPDATESQL
S:ALTERSQL
S:commit
S:MESSAGESQL

CREATESQL:create CREATEQUERY
CREATEQUERY:table TABLENAME ( LISTDEFINE )
CREATEQUERY:tablespace TABLESPACENAME

DROPSQL:drop DROPQUERY
DROPQUERY:table TABLENAME
DROPQUERY:tablespace TABLESPACENAME

SELECTSQL:SUBQUERY UNION_CLAUSE+ ORDERBY LISTQUERY
UNION_CLAUSE:union SUBQUERY
UNION_CLAUSE:ε
SUBQUERY:select LISTQUERY from TABLEQUERY WHERE_CLAUSE+
LISTQUERY:LIST_ALL_NAME
LISTQUERY:*
TABLEQUERY:TABLE_ALL_NAME NEW_TABLENAME+
TABLEQUERY:( SELECTSQL )
WHERE_CLAUSE:where BOOLEAN_EXPRESSION BOOLEAN_EXPRESSIONS*
BOOLEAN_EXPRESSION:QUANTIFIER|LIST_ALL_NAME = QUANTIFIER
BOOLEAN_EXPRESSIONS:, BOOLEAN_EXPRESSION
BOOLEAN_EXPRESSIONS:and BOOLEAN_EXPRESSION
BOOLEAN_EXPRESSIONS:or BOOLEAN_EXPRESSION
BOOLEAN_EXPRESSION:ε

INSERTSQL:insert into TABLEQUERY INSERT_LIST_CLAUSE+ values INSERT_VARIABLE_CLAUSE+
INSERT_LIST_CLAUSE:( LISTNAME LIST_CLAUSE )
LIST_CLAUSE:, LISTNAME
INSERT_VARIABLE_CLAUSE:( QUANTIFIER QUANTIFIERS* )

DELETESQL:delete from TABLENAME WHERE_CLAUSE

UPDATESQL:update TABLENAME set LISTNAME=QUANTIFIER SET_CLAUSE*
SET_CLAUSE:, LISTNAME=QUANTIFIER

ALTERSQL:alter table TABLENAME CHANGE_TABLE_QUERY
CHANGE_TABLE_QUERY:add column NormalTAG QUANTIFIER_TYPE
CHANGE_TABLE_QUERY:del column NormalTAG
