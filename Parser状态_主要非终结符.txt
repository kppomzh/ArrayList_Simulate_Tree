文法
开始符号设置为S，推导用:代替箭头
全大写是非终结符，含有小写的都是终结符，特殊情况是标记常量或变量的QUANTIFIER，里面的终结符也是大写
文法不允许正则表达式之类的写法，必须采用右递归和ε符号表示循环和结束
对于需要TABLE_ALL_NAME和LIST_ALL_NAME的地方，需要在语法解析里加入特殊的补全子程序。因为直接写次一级的文法符号，例如TABLENAME，对于LL(1)的分析器会造成冲突，
这个地方本质上就不是LL的文法。

S:CREATESQL
S:DROPSQL
S:SELECTSQL
S:INSERTSQL
S:DELETESQL
S:UPDATESQL
S:ALTERSQL
S:commit
S:MESSAGESQL

CREATESQL:create CREATEQUERY
CREATEQUERY:table TABLE_ALL_NAME ( LISTDEFINE LISTDEFINES )
CREATEQUERY:tablespace TABLESPACENAME
LISTDEFINE:LISTNAME QUANTIFIER_TYPE LISTDEFINE_LENGTH
LISTDEFINE_LENGTH:( INTEGER )
LISTDEFINE_LENGTH:ε
LISTDEFINES:, LISTDEFINE LISTDEFINES
LISTDEFINES:ε

DROPSQL:drop DROPQUERY
DROPQUERY:table TABLE_ALL_NAME
DROPQUERY:tablespace TABLESPACENAME

SELECTSQL:SUBQUERY UNION_CLAUSE ORDERBY_CLAUSE
UNION_CLAUSE:union SUBQUERY
UNION_CLAUSE:ε
SUBQUERY:select LISTQUERY LISTQUERYS from TABLEQUERY WHERE_CLAUSE
LISTQUERY:LIST_ALL_NAME
LISTQUERY:FRONTTABLE *
FRONTTABLE:TABLE_ALL_NAME .
FRONTTABLE:ε
LISTQUERYS:, LISTQUERY LISTQUERYS
LISTQUERYS:ε
TABLEQUERY:TABLE_ALL_NAME NEW_TABLENAME
TABLEQUERY:( SELECTSQL ) NEW_TABLENAME

WHERE_CLAUSE:where BOOLEAN_EXPRESSION BOOLEAN_EXPRESSIONS
WHERE_CLAUSE:ε
BOOLEAN_EXPRESSION:QUANTIFIER|LIST_ALL_NAME = QUANTIFIER
BOOLEAN_EXPRESSIONS:, BOOLEAN_EXPRESSION BOOLEAN_EXPRESSIONS
BOOLEAN_EXPRESSIONS:and BOOLEAN_EXPRESSION BOOLEAN_EXPRESSIONS
BOOLEAN_EXPRESSIONS:or BOOLEAN_EXPRESSION BOOLEAN_EXPRESSIONS
BOOLEAN_EXPRESSIONS:ε

GROUPBY_CLAUSE:group by LISTQUERY LISTQUERYS
ORDERBY_CLAUSE:order by LIST_ALL_NAME ORDERBY_OPTION
ORDERBY_OPTION:desc
ORDERBY_OPTION:asc

INSERTSQL:insert into TABLEQUERY INSERT_LIST_CLAUSE VALUE_FROM
INSERT_LIST_CLAUSE:( LISTNAME LIST_CLAUSE )
INSERT_LIST_CLAUSE:ε
LIST_CLAUSE:, LISTNAME LIST_CLAUSE
LIST_CLAUSE:ε
VALUE_FROM:values INSERT_VARIABLE_CLAUSE
# VALUE_FROM:SUBQUERY
INSERT_VARIABLE_CLAUSE:( QUANTIFIER QUANTIFIERS )

DELETESQL:delete from TABLE_ALL_NAME WHERE_CLAUSE

UPDATESQL:update TABLE_ALL_NAME set LISTNAME=QUANTIFIER SET_CLAUSE
SET_CLAUSE:, LISTNAME = QUANTIFIER SET_CLAUSE
SET_CLAUSE:ε

ALTERSQL:alter table TABLENAME CHANGE_TABLE_QUERY
CHANGE_TABLE_QUERY:add column LISTNAME QUANTIFIER_TYPE
CHANGE_TABLE_QUERY:del column LISTNAME
